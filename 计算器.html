        <label>平台佣金 (%):</label>

        <input type="number" id="commission" placeholder="输入平台佣金比例">

        

        <label>达人佣金 (%):</label>

        <input type="number" id="influencer" placeholder="输入达人佣金比例">

        

        <label>总收入 (CNY):</label>

        <input type="number" id="revenue" placeholder="输入总收入">

        

        <button onclick="calculateProfit()">计算利润</button>

        

        <div class="result" id="result"></div>

    </div>


    <script>

        function calculateProfit() {

            let cost = parseFloat(document.getElementById('cost').value) || 0;

            let shipping = parseFloat(document.getElementById('shipping').value) || 0;

            let commission = parseFloat(document.getElementById('commission').value) || 0;

            let influencer = parseFloat(document.getElementById('influencer').value) || 0;

            let revenue = parseFloat(document.getElementById('revenue').value) || 0;

            

            let totalCost = cost + shipping + (revenue * (commission / 100)) + (revenue * (influencer / 100));

            let profit = revenue - totalCost;

            let profitMargin = ((profit / revenue) * 100).toFixed(2);

            let roi = ((profit / totalCost) * 100).toFixed(2);

            

            document.getElementById('result').innerHTML = `利润: ${profit.toFixed(2)} CNY (利润率: ${profitMargin}%)<br>ROI: ${roi}%`;

        }

    </script>
